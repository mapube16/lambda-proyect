## Objective

Implement the **4-way debtor classifier** and **6-state collections machine** that powers real-time decisions. By end of this plan:
- When a call reaches the agent, transcript is classified into {puede_pagar|quiere_pagar|no_puede|rechaza}
- Decision engine maps classification to recommended action (immediate offer, payment plan, escalate)
- Debtor transitions through state machine based on classification and commitment
- System tracks confidence levels; escalates to human when confidence < 60%
- Full decision log maintained for compliance & analysis

---

## Context

The debtor classification is the critical path: it determines whether the next step is a sales pitch (for puede_pagar), a gentle reminder (quiere_pagar), a hardship plan (no_puede), or escalation to human (rechaza).

This plan focuses on:
- LLM-based classification with low temperature (0.3) for determinism
- State transitions with validation (reuse Phase 2 state_machine pattern)
- Confidence thresholding for human escalation
- Comprehensive decision logging for audits

async def route_action(
    classification_result: dict,
    debtor_id: str,
    call_uuid: str,
    user_id: str,
) -> dict:
    """
    Route to appropriate next action based on classification.
    Updates debtor estado and call_session in MongoDB.
    
    Returns:
        {
            "action": str,  # immediate_offer | payment_plan | escalate | terminate
            "next_estado": str,  # State transition for debtor
            "message": str,  # What to say next
            "escalation_reason": Optional[str],
        }
    """
    from database import get_db
    
    db = get_db()
    classification = classification_result.get("classification", "error")
    recommended = classification_result.get("recommended_action", "escalate_human")
    
    action_map = {
        "puede_pagar": ("immediate_offer", "propuesta", 
                       "Excelente, le presento nuestro plan de pago inmediato."),
        "quiere_pagar": ("payment_plan", "propuesta",
                        "Entiendo, le presento un plan flexible en cuotas."),
        "no_puede": ("payment_plan", "propuesta",
                    "Entiendo su situación. Ofrecemos un plan a medida."),
        "rechaza": ("escalate", "archivado_collections",
                   "Entiendo, escalando con nuestro equipo."),
        "error": ("escalate", "inicial",
                 "Necesitamos revisión manual."),
    }
    
    action, next_estado, message = action_map.get(
        classification, 
        ("escalate_human", "inicial", "Error: classification unrecognized")
    )
    
    # If confidence too low, always escalate regardless
    if classification_result.get("confidence", 1.0) < 0.6:
        action, next_estado = "escalate_human", "inicial"
        message = "Confianza insuficiente — escalando a agente humano."
    
    # Update debtor estado
    await db.debtors.update_one(
        {"_id": ObjectId(debtor_id), "user_id": user_id},
        {
            "$set": {
                "estado": next_estado,
                "classification": classification,
                "last_call_attempt": datetime.now(timezone.utc),
                "estado_updated_at": datetime.now(timezone.utc),
            }
        },
    )
    
    # Log in call session
    await db.call_sessions.update_one(
        {"call_uuid": call_uuid, "user_id": user_id},
        {
            "$set": {
                "state": "decision_made",
                "outcome": action,
            },
            "$push": {
                "decision_log": {
                    "timestamp": datetime.now(timezone.utc),
                    "action": action,
                    "classification": classification,
                    "confidence": classification_result.get("confidence"),
                }
            },
        },
    )
    
    logger.info(f"[collections] Action routed: {action} for debtor {debtor_id}")
    
    return {
        "action": action,
        "next_estado": next_estado,
        "message": message,
        "escalation_reason": None if action != "escalate_human" else "Low confidence",
    }

@app.post("/webhooks/vapi/call-event")
async def vapi_webhook_handler(payload: VapiWebhookPayload, request: Request):
    """
    POST webhook from VAPI when call events occur.
    On "message" event (transcript received), classify debtor and decide action.
    """
    from landa.agents.collections import classify_debtor, route_action, generate_offer
    
    logger.info(f"[collections] VAPI webhook: event={payload.event}, call_id={payload.call_id}")
    
    db = get_db()
    
    try:
        call_session = await db.call_sessions.find_one({"call_uuid": payload.call_id})
        if not call_session:
            logger.warning(f"[collections] Call session not found: {payload.call_id}")
            return {"status": "error", "reason": "call_not_found"}
        
        user_id = call_session.get("user_id")
        debtor_id = call_session.get("debtor_id")
        
        if payload.event == "initialized":
            # Call started
            await db.call_sessions.update_one(
                {"call_uuid": payload.call_id},
                {"$set": {"state": "initialized", "started_at": datetime.now(timezone.utc)}},
            )
            logger.info(f"[collections] Call initialized: {payload.call_id}")
        
        elif payload.event == "message" and payload.transcript:
            # STT transcript received — classify debtor
            transcript = payload.transcript
            
            # Get debtor info
            debtor = await db.debtors.find_one({"_id": ObjectId(debtor_id)})
            if not debtor:
                logger.error(f"[collections] Debtor not found: {debtor_id}")
                return {"status": "error", "reason": "debtor_not_found"}
            
            # Classify
            classification_result = await classify_debtor(
                debtor_name=debtor.get("name"),
                amount_owed=debtor.get("amount_owed"),
                call_transcript=transcript,
                user_id=user_id,
            )
            
            # Route action
            action_result = await route_action(
                classification_result,
                debtor_id,
                payload.call_id,
                user_id,
            )
            
            # Generate offer if applicable
            if action_result["action"] in ("immediate_offer", "payment_plan"):
                offer = await generate_offer(
                    debtor.get("name"),
                    debtor.get("amount_owed"),
                    classification_result.get("classification"),
                    user_id,
                )
                # Send to VAPI for voice delivery (stub for now)
                logger.info(f"[collections] Offer generated: {offer['offer_type']}")
        
        elif payload.event == "ended":
            # Call terminated
            await db.call_sessions.update_one(
                {"call_uuid": payload.call_id},
                {
                    "$set": {
                        "state": "ended",
                        "ended_at": datetime.now(timezone.utc),
                    }
                },
            )
            logger.info(f"[collections] Call ended: {payload.call_id}")
        
        return {"status": "ok", "call_id": payload.call_id}
    
    except Exception as e:
        logger.error(f"[collections] Webhook error: {e}", exc_info=True)
        raise HTTPException(status_code=500, detail=f"Webhook processing failed: {str(e)}")
