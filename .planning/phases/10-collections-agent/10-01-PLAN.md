- Agent state transitions trigger WebSocket broadcasts to clients (existing pattern from Phase 2)
- System is ready to dispatch agent decision logic in Wave 2

---

## Context

VAPI sends HTTP POST events to your backend whenever:
1. A call is initialized (caller connects)
2. Messages arrive (STT transcript received; debtor speech)
3. Tools are called (agent decides to verify, offer, escalate)
4. Call ends (disconnection)

This plan captures all events, validates them, and writes them to a durable audit log. Webhook authentication via VAPI-signed headers prevents spoofing.

```python
@app.post("/webhooks/vapi/call-event")
async def vapi_webhook_handler(payload: VapiWebhookPayload, request: Request):
    """
    POST webhook from VAPI when call events occur.
    Validates VAPI signature header, persists event, broadcasts to connected clients.
    """
    # Extract VAPI signature for verification (if available)
    signature = request.headers.get("X-Vapi-Signature", "")
    
    # TODO Phase 10: Implement VAPI signature verification using VAPI_WEBHOOK_SECRET
    # For MVP, log the webhook and proceed
    
    logger.info(f"[collections] VAPI webhook: event={payload.event}, call_id={payload.call_id}")
    
    db = get_db()
    
    try:
        if payload.event == "initialized":
            # Call just started — create call_session document
            doc = {
                "call_uuid": payload.call_id,
                "state": "initialized",
                "started_at": datetime.now(timezone.utc),
                "decision_log": [],
                "transcript": "",
                "compliance_flags": [],
                "created_at": datetime.now(timezone.utc),
                "updated_at": datetime.now(timezone.utc),
            }
            result = await db.call_sessions.insert_one(doc)
            logger.info(f"[collections] Created call_session: {result.inserted_id}")
            
        elif payload.event in ("message", "tool_call"):
            # Update transcript or log decision
            await db.call_sessions.update_one(
                {"call_uuid": payload.call_id},
                {
                    "$set": {"updated_at": datetime.now(timezone.utc)},
                    "$push": {
                        "decision_log": {
                            "timestamp": datetime.now(timezone.utc),
                            "event": payload.event,
                            "transcript": payload.transcript or "",
                            "tool_call": payload.tool_call or {},
                        }
                    },
                },
            )
            
        elif payload.event == "ended":
            # Call terminated — finalize
            await db.call_sessions.update_one(
                {"call_uuid": payload.call_id},
                {
                    "$set": {
                        "state": "ended",
                        "ended_at": datetime.now(timezone.utc),
                        "updated_at": datetime.now(timezone.utc),
                    },
                },
            )
            logger.info(f"[collections] Call ended: {payload.call_id}")
        
        return {"status": "ok", "call_id": payload.call_id}
        
    except Exception as e:
        logger.error(f"[collections] Webhook processing error: {e}")
        raise HTTPException(status_code=500, detail="Webhook processing failed")
```

```python
async def classify_debtor(
    debtor_name: str,
    amount_owed: float,
    call_transcript: str,
    user_id: str,
) -> dict:
    """
    Classify a debtor based on their conversation.
    
    Args:
        debtor_name: Name from debtor record
        amount_owed: Amount in USD equiv
        call_transcript: Full transcript from STT
        user_id: Client owner ID (for loading their policy)
    
    Returns:
        {
            "classification": str,
            "confidence": float,
            "reasoning": str,
            "recommended_action": str,

    client = await db.users.find_one({"_id": ObjectId(user_id)})
```
