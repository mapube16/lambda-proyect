**Step A — Write hive_adapter.py:**

```python
"""
HiveAdapter — single seam between FastAPI and aden-hive/hive framework.

ARCHITECTURE RULE: Only this file may import from `framework.*`.
main.py imports HiveAdapter only — never AgentRunner, EventBus, or GraphExecutor directly.
"""
import asyncio
from pathlib import Path
from typing import Callable, Awaitable

from framework.runner.runner import AgentRunner
from framework.runtime.event_bus import EventBus, EventType
from models import AgentState


def _event_to_agent_state(event) -> AgentState | None:
    """Map an AgentEvent to the corresponding AgentState for WebSocket delivery."""
    mapping = {
        EventType.NODE_LOOP_STARTED:       AgentState.THINKING,
        EventType.TOOL_CALL_STARTED:       AgentState.TOOL_USE,
        EventType.TOOL_CALL_COMPLETED:     AgentState.THINKING,   # back to thinking after tool
        EventType.NODE_LOOP_COMPLETED:     AgentState.WAITING,
        EventType.CLIENT_INPUT_REQUESTED:  AgentState.WAITING,    # HITL pause
        EventType.EXECUTION_COMPLETED:     AgentState.IDLE,
        EventType.EXECUTION_FAILED:        AgentState.ERROR,
    }
    return mapping.get(event.type)


class HiveAdapter:
    """
    Adapter between FastAPI ConnectionManager and aden-hive/hive framework.

    Owns all Hive imports. Receives a send_to_user callback from main.py
    so it never needs to import ConnectionManager.

    Tenant isolation: one AgentRunner per user_id, per run.
    SharedMemory isolation: each AgentRunner.run() call creates a separate
    SharedMemory instance internally — no namespace= parameter needed.
    """

    def __init__(self, send_to_user_callback: Callable[[str, dict], Awaitable[None]]):
        # send_to_user_callback: async (user_id: str, message: dict) -> None
        self._send_to_user = send_to_user_callback
        self._runners: dict[str, AgentRunner] = {}  # keyed by user_id

    async def start_run(self, user_id: str, inputs: dict) -> str:
        """
        Create a per-user AgentRunner, subscribe to its EventBus events,
        and launch the run as a background task.

        Returns user_id as the run_id placeholder (Phase 2 — no UUID generation yet).
        """
        from hive_graph import build_stub_graph, build_stub_goal

        runner = AgentRunner(
            agent_path=Path(f"/tmp/hive_stub/{user_id}"),
            graph=build_stub_graph(),
            goal=build_stub_goal(),
            mock_mode=True,
            storage_path=Path(f"/tmp/hive_runs/{user_id}"),
            interactive=False,
            skip_credential_validation=True,
        )

        # Create our EventBus and inject BEFORE calling run().
        # runner._agent_runtime is None until _setup() is called, so inject now.
        our_bus = EventBus()
        our_bus.subscribe(
            event_types=[
                EventType.NODE_LOOP_STARTED,
                EventType.NODE_LOOP_COMPLETED,
                EventType.TOOL_CALL_STARTED,
                EventType.TOOL_CALL_COMPLETED,
                EventType.CLIENT_INPUT_REQUESTED,
                EventType.EXECUTION_COMPLETED,
                EventType.EXECUTION_FAILED,
            ],
            handler=self._make_event_handler(user_id),
            filter_stream=user_id,
        )
        runner._setup(event_bus=our_bus)

        self._runners[user_id] = runner

        # Non-blocking: run in background task so FastAPI event loop is not blocked
        asyncio.create_task(runner.run(inputs=inputs))

        return user_id  # Phase 2: run_id == user_id (placeholder)

    def _make_event_handler(self, user_id: str):
        """Factory: returns async handler that maps AgentEvent → AgentState → WS message."""
        async def handler(event) -> None:
            state = _event_to_agent_state(event)
            if state is not None:
                await self._send_to_user(user_id, {
                    "type": "agent_update",
                    "state": state.value,
                    "node_id": getattr(event, "node_id", None),
                })
        return handler
```
