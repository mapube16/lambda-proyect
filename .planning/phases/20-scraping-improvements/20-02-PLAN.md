def html_to_compressed_markdown(html: str) -> str:
    """
    Strip noise tags, then convert HTML to Markdown for LLM analysis.
    Uses markdownify==1.2.2. Reduces token usage by ~80% vs raw HTML.
    Tables stripped — too verbose for prospecting; links stripped — href not needed by LLM.
    """
    soup = BeautifulSoup(html, "html.parser")
    for tag in soup(["script", "style", "nav", "footer", "header",
                     "aside", "form", "noscript", "svg", "img", "iframe",
                     "table"]):
        tag.decompose()
    clean_html = str(soup)
    markdown = md(
        clean_html,
        strip=["a"],          # no href links needed in LLM analysis
        heading_style="ATX",  # ## heading style
        newline_style="spaces",
    )
    markdown = re.sub(r"\n{3,}", "\n\n", markdown).strip()
    return markdown
```

Note: `md` is already imported as `from markdownify import markdownify as md` (added in Plan 01). `re` and `BeautifulSoup` are already imported. No new imports needed here.

def extract_homepage(url: str) -> str:
    """
    Normalize a URL to the probable company homepage.
    Strips known non-homepage subdomains (blog., careers., etc.) and deep paths.
    Uses tldextract for correct PSL-aware domain parsing (.com.co, .co.uk, etc.).
    Falls back to original URL if unable to parse.

    Examples:
      https://blog.acme.com/article/2024/big-news  -> https://acme.com
      https://careers.empresa.com.co/vacante/123   -> https://empresa.com.co
      https://www.acme.com/about/team              -> https://www.acme.com (path dropped)
      https://acme.com/noticias/articulo-123       -> https://acme.com
    """
    from urllib.parse import urlparse

    if not url:
        return url
    if not url.startswith(("http://", "https://")):
        url = f"https://{url}"

    ext = tldextract.extract(url)
    if not ext.domain or not ext.suffix:
        return url  # unparseable — return as-is

    registered = f"{ext.domain}.{ext.suffix}"  # e.g. "empresa.com.co"
    subdomain = (ext.subdomain or "").lower()

    # Determine the effective subdomain (strip leading www.)
    effective_sub = subdomain
    if effective_sub == "www" or effective_sub == "":
        effective_sub = ""
    elif effective_sub.startswith("www."):
        effective_sub = effective_sub[4:]

    if effective_sub in _NON_HOME_SUBDOMAINS:
        return f"https://{registered}"

    # Check for deep blog/article paths using LOW_QUALITY_PATH_MARKERS
    parsed = urlparse(url)
    path = (parsed.path or "/").lower()
    for marker in LOW_QUALITY_PATH_MARKERS:
        if marker in path:
            netloc = parsed.netloc.lower()
            if netloc.startswith("www."):
                netloc = netloc[4:]
            return f"https://{netloc or registered}"

    # No normalization needed
    return url
```
